```java
import java.lang.annotation.*;

@Target(ElementType.METHOD)

public @interface RateLimit {
```

```java
public <T> T invoke(
    ChatClient chatClient,
    String systemPrompt,
    String userPrompt,
    BeanOutputConverter<T> converter,
    ErrorCode errorCode,
    Logger log
) {
    for (int attempt = 1; attempt <= maxAttempts; attempt++) {
        try {
            String content = chatClient.prompt()
                .system(systemPrompt + converter.getFormat())
                .user(userPrompt)
                .call()
                .content();
            return converter.convert(content);
        } catch (Exception e) {
            if (attempt == maxAttempts) throw new BusinessException(errorCode, "AI 结构化输出失败");
            log.warn("结构化输出第{}次尝试失败: {}", attempt, e.getMessage());
        }
    }
    throw new BusinessException(errorCode, "AI 结构化输出失败");
}
```

```java
/**
 * 添加消息到 Stream
 */
public String streamAdd(String streamKey, Map<String, String> data, int maxLen) {
    RStream<String, String> stream = redissonClient.getStream(streamKey, StringCodec.INSTANCE);
    StreamAddArgs<String, String> args = StreamAddArgs.entries(data).maxLen(maxLen);
    StreamMessageId id = stream.add(args);
    return id.toString();
}

/**
 * 创建消费者组（幂等）
 */
public void createStreamGroup(String streamKey, String groupName) {
    RStream<String, String> stream = redissonClient.getStream(streamKey, StringCodec.INSTANCE);
    try {
        stream.createGroup(StreamCreateGroupArgs.name(groupName).makeStream());
    } catch (Exception e) {
        if (!e.getMessage().contains("BUSYGROUP")) throw e;
    }
}

/**
 * 消费 Stream 消息
 */
public void streamConsumeMessages(String streamKey, String groupName, String consumerName,
        int count, long pollMs, java.util.function.Consumer<Map<StreamMessageId, Map<String, String>>> handler) {
    RStream<String, String> stream = redissonClient.getStream(streamKey, StringCodec.INSTANCE);
    Map<StreamMessageId, Map<String, String>> messages = stream.readGroup(
        groupName, consumerName,
        StreamReadGroupArgs.neverDelivered().count(count).timeout(Duration.ofMillis(pollMs))
    );
    if (messages != null && !messages.isEmpty()) {
        handler.accept(messages);
    }
}

/**
 * 确认消息（ACK）
 */
public void streamAck(String streamKey, String groupName, StreamMessageId messageId) {
    RStream<String, String> stream = redissonClient.getStream(streamKey, StringCodec.INSTANCE);
    stream.ack(groupName, messageId);
}
```

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class ResumeServiceImpl implements IResumeService {

    private final ResumeMapper baseMapper;
    private final ResumeAnalysisMapper analysisMapper;
    private final FileParseService fileParseService;
    private final FileStorageService storageService;
    private final ResumeAnalysisProducer analyzeProducer;   // Stream 生产者

    @Override
    @Transactional
    public ResumeUploadResultDTO uploadResume(MultipartFile file) {
        // 1. 校验文件（从 FileValidationService 逻辑内联）
        fileParseService.validate(file, 10 * 1024 * 1024L,
            Set.of("application/pdf", "application/msword",
                   "application/vnd.openxmlformats-officedocument.wordprocessingml.document"));

        // 2. 计算哈希，去重检查
        String fileHash = fileParseService.calculateHash(file);
        Optional<ResumeEntity> existing = baseMapper.findByFileHash(fileHash);
        if (existing.isPresent()) {
            return ResumeUploadResultDTO.duplicate(toListItem(existing.get()));
        }

        // 3. Tika 解析文本
        String rawContent = fileParseService.parseText(file);
        if (rawContent == null || rawContent.isBlank()) {
            throw new BusinessException(ErrorCodeEnum.RESUME_PARSE_FAILED, "无法提取文本，请确认文件不是扫描版");
        }

        // 4. 上传到 MinIO
        String fileKey = storageService.uploadResume(file);

        // 5. 落库（status=0 待解析）
        ResumeEntity entity = ResumeEntity.builder()
            .fileName(file.getOriginalFilename())
            .fileKey(fileKey)
            .fileSize(file.getSize())
            .fileHash(fileHash)
            .rawContent(rawContent)
            .status(0)
            .build();
        baseMapper.insert(entity);

        // 6. 发送 Stream 消息（异步触发 AI 评分）
        analyzeProducer.send(entity.getId(), rawContent);

        log.info("简历上传完成: resumeId={}, fileName={}", entity.getId(), file.getOriginalFilename());
        return ResumeUploadResultDTO.created(toListItem(entity));
    }
}
```

```java
@Component
public class ResumeAnalysisConsumer extends AbstractStreamConsumer<ResumeAnalysisConsumer.Payload> {

    private final ResumeAnalysisServiceImpl analysisService;
    private final ResumeMapper resumeMapper;

    // 构造器注入...

    record Payload(Long resumeId, String content) {}

    @Override protected String streamKey()       { return StreamConstants.RESUME_ANALYZE_STREAM; }
    @Override protected String groupName()       { return StreamConstants.RESUME_ANALYZE_GROUP; }
    @Override protected String consumerPrefix()  { return StreamConstants.RESUME_ANALYZE_CONSUMER; }
    @Override protected String threadName()      { return "resume-analyze-consumer"; }
    @Override protected String taskDisplayName() { return "简历分析"; }

    @Override
    protected Payload parsePayload(StreamMessageId id, Map<String, String> data) {
        String resumeIdStr = data.get(StreamConstants.FIELD_ID);
        String content     = data.get(StreamConstants.FIELD_CONTENT);
        if (resumeIdStr == null || content == null) return null;
        return new Payload(Long.parseLong(resumeIdStr), content);
    }

    @Override
    protected void markProcessing(Payload p) {
        updateStatus(p.resumeId(), 1, null);   // status=1 解析中
    }

    @Override
    protected void processBusiness(Payload p) {
        if (!resumeMapper.exists(new LambdaQueryWrapper<ResumeEntity>()
                .eq(ResumeEntity::getId, p.resumeId()))) {
            log.warn("简历已删除，跳过分析: resumeId={}", p.resumeId());
            return;
        }
        analysisService.doAnalyze(p.resumeId(), p.content());   // 调用 AI 分析
        updateStatus(p.resumeId(), 2, null);    // status=2 已完成
    }

    @Override
    protected void onProcessFailed(Payload p, Exception e) {
        updateStatus(p.resumeId(), 3, e.getMessage());  // status=3 失败
    }

    private void updateStatus(Long resumeId, int status, String error) {
        ResumeEntity entity = resumeMapper.selectById(resumeId);
        if (entity != null) {
            entity.setStatus(status);
            if (error != null) entity.setAnalyzeError(error.length() > 500 ? error.substring(0, 500) : error);
            resumeMapper.updateById(entity);
        }
    }

    @Override protected String payloadIdentifier(Payload p) { return "resumeId=" + p.resumeId(); }
}
```

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class KnowledgeVectorServiceImpl implements KnowledgeVectorService {

    private final VectorStore vectorStore;          // Spring AI 自动注入 PgVectorStore
    private final KnowledgeBaseMapper kbMapper;

    @Override
    public void vectorize(Long kbId, String content) {
        KnowledgeBaseEntity kb = kbMapper.selectById(kbId);
        if (kb == null) throw new BusinessException(ErrorCode.KNOWLEDGE_BASE_NOT_FOUND);

        // 文本切块
        List<String> chunks = splitIntoChunks(content, 512, 64);   // 512 token，64 重叠

        // 构建 Document 列表
        List<Document> docs = chunks.stream()
            .map(chunk -> {
                Document doc = new Document(chunk);
                doc.getMetadata().put("kbId", kbId.toString());
                doc.getMetadata().put("kbName", kb.getName());
                return doc;
            })
            .toList();

        // 写入 pgvector（Spring AI 自动向量化 + 存储）
        vectorStore.add(docs);

        // 更新状态和分块数
        kb.setStatus(2);
        kb.setChunkCount(chunks.size());
        kbMapper.updateById(kb);
        log.info("知识库向量化完成: kbId={}, chunks={}", kbId, chunks.size());
    }

    @Override
    public void deleteVectors(Long kbId) {
        // 按元数据过滤删除
        vectorStore.delete(List.of(kbId.toString()));
    }
}
```

```java
    List<Document> docs = vectorStore.similaritySearch(
        SearchRequest.query(question)
            .withTopK(5)
            .withFilterExpression("kbId == '" + kbId + "'")
    );

    // 2. 构建上下文
    String context = docs.stream()
        .map(Document::getContent)
        .collect(Collectors.joining("\n\n---\n\n"));

    // 3. 查询对话历史（用于多轮）
    List<RagChatMessageDTO> history = getHistory(sessionId);

    // 4. 调用 LLM
    ChatClient client = registry.getChatClientOrDefault(null);
    String systemPrompt = loadPrompt("knowledge/rag-system.st");
    String userPrompt   = loadPrompt("knowledge/rag-user.st")
        .replace("{context}", context)
        .replace("{question}", question);

    String answer = client.prompt()
        .system(systemPrompt)
        .user(userPrompt)
        .call()
        .content();

    // 5. 持久化 user/assistant 消息
    saveMessage(sessionId, "user", question);
    saveMessage(sessionId, "assistant", answer);

    return new RagQueryResponse(answer, docs.stream()
        .map(d -> new SourceReference((String) d.getMetadata().get("kbId"), d.getContent()))
        .toList());
}
```
